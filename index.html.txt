<script>

let total=0
let shots=[]
let rounds=[]

function addScore(){

let hole=document.getElementById("hole").value
let strokes=parseInt(document.getElementById("strokes").value)

let table=document.getElementById("scoreTable")

let row=table.insertRow()

row.insertCell(0).innerHTML=hole
row.insertCell(1).innerHTML=strokes

total+=strokes

document.getElementById("total").innerText=total

}

function addClub(){

let club=document.getElementById("club").value
let yard=document.getElementById("yard").value

let table=document.getElementById("clubTable")

let row=table.insertRow()

row.insertCell(0).innerHTML=club
row.insertCell(1).innerHTML=yard

}

function addHole(){

let hole=document.getElementById("holeY").value
let yard=document.getElementById("distY").value

let table=document.getElementById("courseTable")

let row=table.insertRow()

row.insertCell(0).innerHTML=hole
row.insertCell(1).innerHTML=yard

}

function addShot(){

let club=document.getElementById("shotClub").value
let dist=parseInt(document.getElementById("shotDist").value)

shots.push(dist)

let table=document.getElementById("shotTable")

let row=table.insertRow()

row.insertCell(0).innerHTML=club
row.insertCell(1).innerHTML=dist

let avg=shots.reduce((a,b)=>a+b)/shots.length

document.getElementById("avgDist").innerText=Math.round(avg)

}

function addRound(){

let score=parseInt(document.getElementById("roundScore").value)

rounds.push(score)

let avg=rounds.reduce((a,b)=>a+b)/rounds.length

let handicap=Math.round((avg-72)*0.8)

document.getElementById("handicap").innerText=handicap

}
